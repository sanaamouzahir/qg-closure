\section{Governing equations}

\begin{align}
&\partial_t\omega = L\omega + N,\qquad N = -J(\psi,\omega)+F,\qquad
\psi = \nabla^{-2}\omega,\qquad \dot F = 0,\\
&\hat L(\mathbf{k}) = -\nu\kappa^2-\mu + i\beta k_x/\kappa^2,\qquad
\kappa=|\mathbf{k}|,\qquad
J(a,b)=\partial_x a\,\partial_y b-\partial_y a\,\partial_x b,\\
&\omega^{(k)} = L\,\omega^{(k-1)}+N^{(k-1)},\qquad
\psi^{(k)}=\nabla^{-2}\omega^{(k)},\\
&N^{(m)} = -\sum_{j=0}^{m}\binom{m}{j}\,
J\!\bigl(\psi^{(m-j)},\,\omega^{(j)}\bigr),\qquad m\ge1.
\end{align}

\section{Input stack and finite-difference estimates}

\begin{align}
&X=\bigl[\omega_0,\omega_{-1},\dots,\omega_{-(S-1)},\;
\psi_0,\psi_{-1},\dots,\psi_{-(S-1)}\bigr],\qquad
\omega_{-i}=\omega(t-i\DT),\quad S=7,\\
&\sum_{i=0}^{S-1}W_{k i}\,\frac{(-i)^{p}}{p!}=\delta_{kp},
\quad p=0,\dots,S-1
\qquad(\text{Vandermonde rows }W),\\
&\omega^{(k)}_{\mathrm{FD}}=\frac{1}{\DT^{\,k}}\sum_{i=0}^{S-1}
W_{k i}\,\omega_{-i},\qquad
\psi^{(k)}_{\mathrm{FD}}=\frac{1}{\DT^{\,k}}\sum_{i=0}^{S-1}
W_{k i}\,\psi_{-i},\\
&\varepsilon_k \equiv \omega^{(k)}_{\mathrm{FD}}-\omega^{(k)}
= C_k\,\DT^{\,S-k}\,\omega^{(S)}+O(\DT^{\,S-k+1}),\qquad
C_k = -\frac{1}{S!}\sum_{i=0}^{S-1}W_{k i}\,(-i)^{S},\\
&\psi^{(k)}_{\mathrm{FD}}-\psi^{(k)}
= C_k\,\DT^{\,S-k}\,\nabla^{-2}\omega^{(S)}+O(\DT^{\,S-k+1}).
\end{align}

\section{Learned spatial operators and the error at the $N$-level}

\begin{align}
&D^x_\theta=\partial_x+\delta^x,\qquad D^y_\theta=\partial_y+\delta^y,\qquad
\delta=(\delta^x,\delta^y)\ \text{local, one pair for the pool},\\
&\mathcal{P}=\{\text{members}\times\DT\times\text{anchors}\},\\
&N^{(m)}_{\mathrm{est}}
=-\sum_{j=0}^{m}\binom{m}{j}\,
J_\theta\!\bigl(\psi^{(m-j)}_{\mathrm{FD}},\,\omega^{(j)}_{\mathrm{FD}}\bigr),\\
&J_\theta(a,b)=D^x_\theta a\,D^y_\theta b-D^y_\theta a\,D^x_\theta b,\\
&\Delta N^{(m)}\equiv N^{(m)}_{\mathrm{est}}-N^{(m)}
= T^{(m)}+\mathcal{A}^{(m)}[\delta]
+O(\varepsilon\delta,\ \varepsilon^2,\ \delta^2),\\
&T^{(m)}=-\sum_{j=0}^{m}\binom{m}{j}
\Bigl[J\!\bigl(\nabla^{-2}\varepsilon_{m-j},\,\omega^{(j)}\bigr)
+J\!\bigl(\psi^{(m-j)},\,\varepsilon_{j}\bigr)\Bigr],\\
&\mathcal{A}^{(m)}[\delta]
=-\sum_{j=0}^{m}\binom{m}{j}\,\delta J_{m-j,\,j},\\
&\delta J_{i j}
=\delta^x\psi^{(i)}\,\partial_y\omega^{(j)}
+\partial_x\psi^{(i)}\,\delta^y\omega^{(j)}\notag\\
&\quad
-\delta^y\psi^{(i)}\,\partial_x\omega^{(j)}
-\partial_y\psi^{(i)}\,\delta^x\omega^{(j)}.
\end{align}
Distribution of the time error through the binomial (leading powers of $\DT$,
via (8)--(9)):
\begin{align}
T^{(m)}
&=-\sum_{j=0}^{m}\binom{m}{j}\DT^{\,S-(m-j)}\,C_{m-j}\,
J\!\bigl(\nabla^{-2}\omega^{(S)},\,\omega^{(j)}\bigr)\notag\\
&\qquad
-\sum_{j=0}^{m}\binom{m}{j}\DT^{\,S-j}\,C_{j}\,
J\!\bigl(\psi^{(m-j)},\,\omega^{(S)}\bigr)+\text{h.o.t.}\\
&=\DT^{\,S-m}\Bigl[
-C_m\,J\!\bigl(\nabla^{-2}\omega^{(S)},\,\omega^{(0)}\bigr)
-C_m\,J\!\bigl(\psi^{(0)},\,\omega^{(S)}\bigr)\Bigr]
+O(\DT^{\,S-m+1}).
\end{align}

\section{Pooled least squares and the unconditioned Wiener floor}

\begin{align}
&\mathcal{L}(\delta)=\sum_{m=1}^{3}\E_{\mathcal{P}}
\bigl\lVert T^{(m)}+\mathcal{A}^{(m)}[\delta]\bigr\rVert^{2},
\qquad T=(T^{(1)},T^{(2)},T^{(3)}),\quad
\mathcal{A}[\delta]=(\mathcal{A}^{(1)},\mathcal{A}^{(2)},\mathcal{A}^{(3)}),\\
&\E_{\mathcal{P}}\bigl\langle
\mathcal{A}[\eta],\,T+\mathcal{A}[\delta^\star]\bigr\rangle=0
\quad\forall\,\eta\in\mathcal{F}
\qquad(\mathcal{F}=\text{width-}W\text{ filter pairs}),\\
&\mathcal{A}[\delta^\star]=-\,\Pi_{\mathcal{A}\mathcal{F}}\,T,\qquad
\mathcal{L}_{\min}
=\E_{\mathcal{P}}\bigl\lVert(I-\Pi_{\mathcal{A}\mathcal{F}})\,T
\bigr\rVert^{2}
=\E_{\mathcal{P}}\norm{T}^{2}
-\E_{\mathcal{P}}\norm{\Pi_{\mathcal{A}\mathcal{F}}T}^{2},\\
&\mathcal{L}_{\min}
=\underbrace{\E_{\mathcal{P}}\norm{T_r}^{2}}_{\text{(i) cascade remainder}}
+\underbrace{\E_{\mathcal{P}}\bigl\lVert
\mathcal{A}\bigl[\delta^\star_{(\text{member},\DT)}-\bar\delta^\star\bigr]
\bigr\rVert^{2}}_{\text{(ii) pooled-variance mismatch}}
+\underbrace{\mathrm{def}(\mathcal{F})}_{\text{(iii) finite-width shape deficit}}.
\end{align}
Measured (S=7, $m=2$, width 15, valid band):
\begin{equation}
\text{diag.\ absorbable}=0.21\text{--}0.29,\qquad
\text{full-slot absorbable}=0.36\text{--}0.44,\qquad
\text{(ii)}=0.72,\qquad \text{(iii)}=0.31.
\end{equation}

\section{Single-mode analysis of the driver $\omega^{(S)}$}

Sweeping split ($\mathbf{u}=\mathbf{U}(t)+\mathbf{u}'$):
\begin{align}
&\partial_t\hat\omega(\mathbf{k},t)
=\bigl[\,i\Omega(\mathbf{k},t)+\lambda(\kappa)\bigr]\hat\omega
+\hat f(\mathbf{k},t),\qquad
\Omega=-\,\mathbf{k}\!\cdot\!\mathbf{U}+\frac{\beta k_x}{\kappa^{2}},\qquad
\lambda=-\nu\kappa^{2}-\mu,\\
&a\equiv i\Omega+\lambda,\qquad \dot a=0\ \ (\text{frozen}):\qquad
\hat\omega^{(n)}=a^{n}\hat\omega
+\sum_{j=0}^{n-1}a^{\,j}\hat f^{\,(n-1-j)},\\
&\hat\omega^{(S)}
=a^{\,S-k}\,\hat\omega^{(k)}
+\underbrace{\sum_{j=0}^{S-k-1}a^{\,j}\,
\hat f^{\,(S-1-j)}}_{\displaystyle \hat r_1},\\
&|\Omega|\gg|\lambda|:\qquad
a^{\,S-k}\approx(i\Omega)^{S-k}
=|\Omega|^{S-k}\,e^{\,i\phi},\qquad
\phi=(S-k)\tfrac{\pi}{2}\,\mathrm{sign}(\Omega),\\
&\sigma_\omega(\kappa)^{2}
=\frac{\E\norm{\dot\omega_\kappa}^{2}}{\E\norm{\omega_\kappa}^{2}}
=\bigl\langle\Omega^{2}+\lambda^{2}\bigr\rangle_{\mathrm{shell}}
\approx\bigl\langle\Omega^{2}\bigr\rangle_{\mathrm{shell}},\\
&\bigl\langle\Omega^{2}\bigr\rangle_{\mathrm{shell}}
=\tfrac{1}{2}\kappa^{2}U^{2}
+\tfrac{1}{2}\frac{\beta^{2}}{\kappa^{2}}
-U\beta\cos\alpha_{\mathbf{U}}
\qquad(\alpha_{\mathbf{U}}=\text{angle of }\mathbf{U}\text{ to }x),\\
&\widehat{\omega^{(S)}}_\kappa
=\sigma_\omega(\kappa)^{S-k}\,e^{i\phi(\kappa)}\,
\widehat{\omega^{(k)}}_\kappa+\hat r,\qquad
\hat r=\hat r_1+\hat r_2+\hat r_3.
\end{align}

\section{Remainder taxonomy}

\subsection*{$r_1$ (cascade input; irreducible)}
\begin{align}
&\hat r_1=\sum_{j=0}^{S-k-1}a^{\,j}\,\hat f^{\,(S-1-j)},\qquad
\bigl\langle \hat r_{1},\,\hat\omega^{(k)}\bigr\rangle_{\mathrm{shell}}
\approx 0,\\
&\text{measured incoherent fraction }=1-\mathrm{coh}^{2}
=0.71\text{--}0.79\ \ (\text{healthy tiers}).
\end{align}

\subsection*{$r_2$ (coefficient drift; conditionable in time)}
\begin{align}
&\dot a\neq0:\qquad
\hat\omega^{(n)}
=\Bigl[a^{n}+\tbinom{n}{2}a^{\,n-2}\dot a
+O(\ddot a,\dot a^{2})\Bigr]\hat\omega,\\
&\hat r_2=\Bigl(\tbinom{S}{2}-\tbinom{k}{2}\Bigr)
a^{\,S-k-2}\,\dot a\;\hat\omega^{(k)},\qquad
\tbinom{S}{2}-\tbinom{k}{2}=\frac{(S-k)(S+k-1)}{2},\\
&\dot a=i\dot\Omega=-i\,\mathbf{k}\!\cdot\!\dot{\mathbf{U}},\qquad
\frac{|\hat r_2|}{|a^{S-k}\hat\omega^{(k)}|}
=\frac{(S-k)(S+k-1)}{2}\,\frac{|\dot a|}{|a|^{2}}
\sim\frac{(S-k)(S+k-1)}{2}\,\frac{\tau_\kappa}{T_U},\\
&\tau_\kappa=\frac{1}{\sigma_\omega(\kappa)},\\
&(S,k)=(7,2):\quad
\tbinom{S}{2}-\tbinom{k}{2}=20 .
\end{align}

\subsection*{$r_3$ (intra-shell anisotropy; conditionable in angle)}
\begin{align}
&\Omega(\mathbf{k})=-\kappa U\cos\theta_{\mathbf U}
+\frac{\beta}{\kappa}\cos\theta_x,\qquad
\Omega^{2}=\sigma(\kappa)^{2}\bigl(1+\eta(\hat{\mathbf k})\bigr),\qquad
\bigl\langle\eta\bigr\rangle_{\mathrm{shell}}=0,\\
&\text{sweeping only: }\ \eta=\cos2\theta_{\mathbf U}\in[-1,1],\\
&\hat r_3=\Bigl[(i\Omega)^{S-k}-i^{\,S-k}\sigma(\kappa)^{S-k}\Bigr]
\hat\omega^{(k)},\\
&\frac{|\Omega|^{S-k}}{\sigma^{S-k}}=(1+\eta)^{\frac{S-k}{2}}
\in\bigl[0,\;2^{\frac{S-k}{2}}\bigr],\\
&\cos^{5}\theta
=\frac{10\cos\theta+5\cos3\theta+\cos5\theta}{16},\\
&\bigl\langle\cos^{5}\theta\bigr\rangle_{\theta}=0
\qquad(\text{best isotropic transfer of the sweeping part}=0).
\end{align}

\section{Physics conditioning}

Per-channel factorization of the measured optimal transfer:
\begin{align}
&r_j^{(k)}(\kappa)
=\underbrace{\DT_j^{\,S-k}}_{\text{exact}}\times
\underbrace{g^{(k)}\bigl(\kappa;\,Re,\beta,\mu\bigr)}_{\text{regime shape}}
+\text{h.o.t.},\\
&g^{(k)}(\kappa;Re,\beta,\mu)
=C_k\,\sigma_\omega(\kappa;Re,\beta,\mu)^{S-k}\,e^{i\phi(\kappa)}
+\dots
\qquad(\text{regime enters only through }\sigma_\omega(\kappa)),\\
&x(\kappa)\equiv\DT\,\hat\sigma_\omega(\kappa),\qquad
\hat\sigma_{\mathrm{raw}}(\kappa)
=\frac{\norm{(\hat\omega_0-\hat\omega_{-1})_\kappa}}
{\DT\,\norm{(\hat\omega_0)_\kappa}},\qquad
\hat\sigma(\kappa)
=\frac{2}{\DT}\arcsin\!\Bigl(\min\bigl(\tfrac{\hat\sigma_{\mathrm{raw}}\DT}{2},
\,1\bigr)\Bigr),\\
&\text{measured: }\ \mathrm{med}_\kappa
\frac{|\hat\sigma-\sigma|}{\sigma}=0.000\text{--}0.001,\\
&\text{measured $\DT$-collapse of }r_j/\DT^{S-k}:\ 1.6\times
\ (\text{truncation-dominated tiers}).
\end{align}
Conditioned operator families (per channel $k$, per direction $d$):
\begin{align}

\mathrm{def}(\mathcal{F}_{\mathrm{loc}})=\mathrm{res}(W)
\quad(\text{measured, }W=15:\ 0.31).
\end{align}

\section{Floors with and without conditioning}

\begin{align}
\mathcal{L}_{\min}^{\mathrm{uncond}}
&=\E_{\mathcal{P}}\norm{T_{r_1}}^{2}
+\E_{\mathcal{P}}\norm{\mathcal{A}[\delta_j^\star-\bar\delta^\star]}^{2}
+\mathrm{def}(\mathcal{F})
\qquad(\text{measured: net absorption }\approx0.03),\\
\mathcal{L}_{\min}^{\mathrm{cond}}
&=\E_{\mathcal{P}}\norm{T_{r_1}}^{2}
+\underbrace{\Bigl[\tfrac{(S-1)\DT}{T_U}\Bigr]\text{-residue of }r_2}
_{\text{within-span drift}}
+\underbrace{\text{unmodeled harmonics of }r_3}
_{\substack{\text{isotropic }g:\ \text{all sweeping}\\
+(k_x/\kappa)^2\text{-features}:\ \cos3\theta,\cos5\theta\ \text{only}}}
+\mathrm{def}(\mathcal{F})+\text{h.o.t.},\\
\mathrm{floor}_j^{\mathrm{cond}}
&=\mathrm{raw}_j\sqrt{\ \mathrm{irr}_j
+\mathrm{coh}_j\bigl(\mathrm{def}(\mathcal{F})+\text{h.o.t.}\bigr)\ }
\qquad(\text{per tier }j).
\end{align}
