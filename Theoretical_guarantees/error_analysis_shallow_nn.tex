\section{Governing equations}

\begin{align}
&\partial_t\omega = L\omega + N,\qquad N = -J(\psi,\omega)+F,\qquad
\psi = \nabla^{-2}\omega,\qquad \dot F = 0,\\
&\hat L(\mathbf{k}) = -\nu\kappa^2-\mu + i\beta k_x/\kappa^2,\qquad
\kappa=|\mathbf{k}|,\qquad
J(a,b)=\partial_x a\,\partial_y b-\partial_y a\,\partial_x b.
\end{align}
Derivative recursion; $J$ bilinear, Leibniz on $N=-J(\psi,\omega)+F$,
$F^{(m)}=0$ for $m\ge1$:
\begin{align}
&\omega^{(k)} = \partial_t\,\omega^{(k-1)} = L\,\omega^{(k-1)}+N^{(k-1)},
\qquad \psi^{(k)}=\nabla^{-2}\omega^{(k)},\\
&N^{(m)} = -\partial_t^{\,m} J(\psi,\omega)
= -\sum_{j=0}^{m}\binom{m}{j}\,
J\!\bigl(\psi^{(m-j)},\,\omega^{(j)}\bigr),\qquad m\ge1.
\end{align}

\section{Learned spatial operators and the error at the $N$-level}

\begin{align}
&D^x_\theta=\partial_x+\delta^x,\qquad D^y_\theta=\partial_y+\delta^y,\qquad
\delta=(\delta^x,\delta^y)\ \text{local, one pair for the pool},\\
&\mathcal{P}=\{\text{members}\times\DT\times\text{anchors}\},\qquad

=-\sum_{j=0}^{m}\binom{m}{j}\,
J_\theta\!\bigl(\psi^{(m-j)}_{\mathrm{FD}},\,\omega^{(j)}_{\mathrm{FD}}\bigr).
\end{align}
Expansion of one slot; bilinearity of $J_\theta$ in each argument, then
linearity of $J_\theta$ in the operators:
\begin{align}
J_\theta\bigl(\psi^{(i)}+\varepsilon^{\psi}_{i},\,
\omega^{(j)}+\varepsilon_{j}\bigr)
&=J_\theta\bigl(\psi^{(i)},\omega^{(j)}\bigr)
+J_\theta\bigl(\varepsilon^{\psi}_{i},\omega^{(j)}\bigr)
+J_\theta\bigl(\psi^{(i)},\varepsilon_{j}\bigr)
+J_\theta\bigl(\varepsilon^{\psi}_{i},\varepsilon_{j}\bigr),\\
J_\theta(a,b)
&=(\partial_x+\delta^x)a\,(\partial_y+\delta^y)b
-(\partial_y+\delta^y)a\,(\partial_x+\delta^x)b\notag\\
&=J(a,b)+\delta J(a,b)+O(\delta^2),\\
\delta J(a,b)
&\equiv\delta^x a\,\partial_y b+\partial_x a\,\delta^y b
-\delta^y a\,\partial_x b-\partial_y a\,\delta^x b .
\end{align}
Collection over slots, discarding
$O(\varepsilon\delta,\varepsilon^2,\delta^2)$:
\begin{align}
\Delta N^{(m)}
&\equiv N^{(m)}_{\mathrm{est}}-N^{(m)}
= T^{(m)}+\mathcal{A}^{(m)}[\delta]
+O(\varepsilon\delta,\ \varepsilon^2,\ \delta^2),\\
T^{(m)}
&=-\sum_{j=0}^{m}\binom{m}{j}
\Bigl[J\!\bigl(\nabla^{-2}\varepsilon_{m-j},\,\omega^{(j)}\bigr)
+J\!\bigl(\psi^{(m-j)},\,\varepsilon_{j}\bigr)\Bigr],\\
\mathcal{A}^{(m)}[\delta]
&=-\sum_{j=0}^{m}\binom{m}{j}\,
\delta J\!\bigl(\psi^{(m-j)},\,\omega^{(j)}\bigr).
\end{align}
Distribution of the time error through the binomial: insertion of (10)--(11)
into (18); each slot carries the $\DT$-power of its own channel:
\begin{align}

&=-\sum_{j=0}^{m}\binom{m}{j}\,C_{m-j}\,\DT^{\,S-(m-j)}\,
J\!\bigl(\nabla^{-2}\omega^{(S)},\,\omega^{(j)}\bigr)\notag\\
&\qquad
-\sum_{j=0}^{m}\binom{m}{j}\,C_{j}\,\DT^{\,S-j}\,
J\!\bigl(\psi^{(m-j)},\,\omega^{(S)}\bigr)+\text{h.o.t.}
\end{align}

\section{Pooled least squares and the unconditioned Wiener floor}

\begin{align}
&\mathcal{L}(\delta)=\sum_{j\in\mathcal{J}} w_j\,\mathcal{L}_j(\delta),
\qquad
\mathcal{L}_j(\delta)=\E_j\norm{T+\mathcal{A}[\delta]}^{2},
\qquad
\mathcal{J}=\{(\text{member},\DT)\ \text{tiers}\},\ \textstyle\sum_j w_j=1,\\
&T=(T^{(1)},T^{(2)},T^{(3)}),\qquad
\mathcal{A}[\delta]=(\mathcal{A}^{(1)},\mathcal{A}^{(2)},\mathcal{A}^{(3)}),
\qquad \mathcal{A}\ \text{linear in }\delta .
\end{align}
First variation at the per-tier optimum $\delta^\star_j\in\mathcal{F}$
($\mathcal{F}=$ width-$W$ filter pairs):
\begin{align}
&0=\tfrac{d}{ds}\Big|_{s=0}\,
\E_j\norm{T+\mathcal{A}[\delta^\star_j+s\,\eta]}^{2}
=2\,\E_j\bigl\langle
\mathcal{A}[\eta],\,T+\mathcal{A}[\delta^\star_j]\bigr\rangle
\qquad\forall\,\eta\in\mathcal{F}.
\end{align}
Orthogonal decomposition of any $\delta\in\mathcal{F}$ about
$\delta^\star_j$; the cross term vanishes by (23):
\begin{align}
\mathcal{L}_j(\delta)
&=\E_j\norm{T+\mathcal{A}[\delta^\star_j]
+\mathcal{A}[\delta-\delta^\star_j]}^{2}\notag\\
&=\mathcal{L}_j(\delta^\star_j)
+2\,\E_j\bigl\langle \mathcal{A}[\delta-\delta^\star_j],\,
T+\mathcal{A}[\delta^\star_j]\bigr\rangle
+\E_j\norm{\mathcal{A}[\delta-\delta^\star_j]}^{2}\notag\\
&=\mathcal{L}_j(\delta^\star_j)
+\E_j\norm{\mathcal{A}[\delta-\delta^\star_j]}^{2}.
\end{align}
Pooled functional at the single shared filter; minimization over
$\bar\delta\in\mathcal{F}$:
\begin{align}
\mathcal{L}(\bar\delta)
&=\sum_j w_j\,\mathcal{L}_j(\delta^\star_j)
+\sum_j w_j\,\E_j\norm{\mathcal{A}[\delta^\star_j-\bar\delta]}^{2},\\
\mathcal{L}_{\min}^{\mathrm{uncond}}
&=\underbrace{\sum_j w_j\,\mathcal{L}_j(\delta^\star_j)}
_{\text{per-tier floors}}
+\underbrace{\min_{\bar\delta\in\mathcal{F}}
\sum_j w_j\,\E_j\norm{\mathcal{A}[\delta^\star_j-\bar\delta]}^{2}}
_{\text{(ii) pooled-variance mismatch}}.
\end{align}
Per-tier floor split by enlarging the family:
$\mathcal{F}\subset\mathcal{F}_\infty$
($\mathcal{F}_\infty=$ all shell-diagonal transfers), with
$\delta^{\star\star}_j$ the optimum in $\mathcal{F}_\infty$:
\begin{align}
\mathcal{L}_j(\delta^\star_j)
&=\underbrace{\mathcal{L}_j(\delta^{\star\star}_j)}
_{\text{(i) cascade remainder, tier }j}
+\underbrace{\E_j\norm{\mathcal{A}
[\delta^{\star\star}_j-\delta^{\star}_j]}^{2}}
_{\text{(iii) shape deficit of width }W,\ \text{tier }j},\\
\mathcal{L}_{\min}^{\mathrm{uncond}}
&=\underbrace{\sum_j w_j\,\mathcal{L}_j(\delta^{\star\star}_j)}_{\text{(i)}}
+\underbrace{\text{(ii)}}_{\text{pooled variance}}
+\underbrace{\sum_j w_j\,\E_j\norm{\mathcal{A}

\end{align}

\section{Single-mode analysis of the driver $\omega^{(S)}$}

Fourier transform of (1) at fixed $\mathbf{k}$:
\begin{equation}
\partial_t\hat\omega(\mathbf{k},t)
=\hat L(\mathbf{k})\,\hat\omega(\mathbf{k},t)+\hat N(\mathbf{k},t).
\end{equation}
\textbf{Assumption A1 (sweeping split).}
$\mathbf{u}=\mathbf{U}(t)+\mathbf{u}'$; the dominant advection of mode
$\mathbf{k}$ is transport by $\mathbf{U}$:
\begin{align}
&\hat N(\mathbf{k},t)
=-\,i\bigl(\mathbf{k}\!\cdot\!\mathbf{U}(t)\bigr)\hat\omega(\mathbf{k},t)
+\hat f(\mathbf{k},t)
\qquad(\hat f=\text{genuine triads}),\\
&\partial_t\hat\omega
=\bigl[\,i\Omega(\mathbf{k},t)+\lambda(\kappa)\bigr]\hat\omega
+\hat f,\qquad
\Omega=-\,\mathbf{k}\!\cdot\!\mathbf{U}+\frac{\beta k_x}{\kappa^{2}},\qquad
\lambda=-\nu\kappa^{2}-\mu .
\end{align}
\textbf{Assumption A2 (frozen coefficients).}
$\dot a=0$ over the stencil span, $a\equiv i\Omega+\lambda$.
Induction claim:
\begin{equation}
\hat\omega^{(n)}=a^{n}\hat\omega
+\sum_{j=0}^{n-1}a^{\,j}\hat f^{\,(n-1-j)} .
\end{equation}
Base $n=1$: (31) $=a\hat\omega+\hat f$ $=$ (32) with the single term
$a^{0}\hat f^{(0)}$. Step $n\to n+1$: differentiate (32) under A2,
\begin{align}
\hat\omega^{(n+1)}
&=a^{n}\hat\omega^{(1)}+\sum_{j=0}^{n-1}a^{\,j}\hat f^{\,(n-j)}
=a^{n}\bigl(a\hat\omega+\hat f\bigr)
+\sum_{j=0}^{n-1}a^{\,j}\hat f^{\,(n-j)}\notag\\
&=a^{n+1}\hat\omega
+\underbrace{a^{n}\hat f^{\,(0)}}_{j=n\ \text{term}}
+\sum_{j=0}^{n-1}a^{\,j}\hat f^{\,(n-j)}
=a^{n+1}\hat\omega+\sum_{j=0}^{n}a^{\,j}\hat f^{\,(n-j)}.\qquad\square
\end{align}
Elimination of $\hat\omega$ between orders $S$ and $k$: (32) at $n=S$ and
$n=k$, second multiplied by $a^{S-k}$, subtracted:
\begin{align}
a^{S-k}\hat\omega^{(k)}
&=a^{S}\hat\omega+\sum_{j=0}^{k-1}a^{\,j+S-k}\,\hat f^{\,(k-1-j)}
=a^{S}\hat\omega+\sum_{j'=S-k}^{S-1}a^{\,j'}\,\hat f^{\,(S-1-j')}
\quad(j'=j+S-k),\\
\hat\omega^{(S)}-a^{S-k}\hat\omega^{(k)}
&=\sum_{j=0}^{S-1}a^{\,j}\hat f^{\,(S-1-j)}
-\sum_{j=S-k}^{S-1}a^{\,j}\hat f^{\,(S-1-j)}
=\sum_{j=0}^{S-k-1}a^{\,j}\,\hat f^{\,(S-1-j)},\\
\hat\omega^{(S)}
&=a^{\,S-k}\,\hat\omega^{(k)}
+\underbrace{\sum_{j=0}^{S-k-1}a^{\,j}\,
\hat f^{\,(S-1-j)}}_{\displaystyle \hat r_1}.
\end{align}
\textbf{Assumption A3 (advective band).} $\Omega^{2}\gg\lambda^{2}$:
\begin{align}
a^{\,S-k}
&=(i\Omega)^{S-k}\Bigl(1+\frac{\lambda}{i\Omega}\Bigr)^{S-k}
=(i\Omega)^{S-k}\Bigl(1+(S-k)\frac{\lambda}{i\Omega}

Shell rate identification: from (31) with
\textbf{Assumption A4} ($\hat f$ subdominant in the shell balance of
